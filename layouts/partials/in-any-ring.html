{{- /* True if (x,y) is inside any one of several disconnected polygon rings -- a state's landmass can be split across separate pieces (e.g. Michigan's peninsulas, islands). Params: x, y, rings (slice of rings, each a slice of [x,y] pairs). Returns bool. */ -}} {{- $x := .x -}} {{- $y := .y -}} {{- $rings := .rings -}} {{- $inside := false -}} {{- range $rings }} {{- if not $inside }} {{- $inside = partial "point-in-polygon.html" (dict "x" $x "y" $y "poly" .) }} {{- end }} {{- end }} {{- return $inside -}}
