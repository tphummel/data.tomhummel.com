{{- /* Average of vertices (not a true area-weighted centroid, close enough for nudging a point back inside a polygon). Params: poly (slice of {x,y} dicts -- not raw [x,y] pairs, since Hugo's `append` flattens slice-typed values instead of nesting them). Returns dict x,y. */ -}} {{- $poly := .poly -}} {{- $n := len $poly -}} {{- $sumX := 0.0 -}} {{- $sumY := 0.0 -}} {{- range $poly }} {{- $sumX = add $sumX .x }} {{- $sumY = add $sumY .y }} {{- end }} {{- return (dict "x" (div $sumX $n) "y" (div $sumY $n)) -}}
