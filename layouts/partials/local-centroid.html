{{- /* Average of the k polygon vertices nearest (x,y) -- a "local" centroid that stays close to the relevant piece of coastline, unlike averaging the whole ring (which for an elongated or multi-part state can sit far from where the point actually is). Params: x, y, ring (slice of [x,y] pairs), k. Returns dict x,y. */ -}} {{- $x := .x -}} {{- $y := .y -}} {{- $ring := .ring -}} {{- $k := .k -}} {{- $withDist := slice -}} {{- range $ring }} {{- $px := index . 0 }} {{- $py := index . 1 }} {{- $d := add (mul (sub $px $x) (sub $px $x)) (mul (sub $py $y) (sub $py $y)) }} {{- $withDist = $withDist | append (dict "dist" $d "x" $px "y" $py) }} {{- end }} {{- $sorted := sort $withDist "dist" "asc" -}} {{- $nearest := first $k $sorted -}} {{- $n := len $nearest -}} {{- $sumX := 0.0 -}} {{- $sumY := 0.0 -}} {{- range $nearest }} {{- $sumX = add $sumX .x }} {{- $sumY = add $sumY .y }} {{- end }} {{- return (dict "x" (div $sumX $n) "y" (div $sumY $n)) -}}
