{{- /* Which of several polygon rings has the vertex closest to (x,y). Params: x, y, rings (slice of rings, each a slice of [x,y] pairs). Returns that ring. */ -}} {{- $x := .x -}} {{- $y := .y -}} {{- $rings := .rings -}} {{- $bestDist := 1.0e18 -}} {{- $bestRing := index $rings 0 -}} {{- range $ring := $rings }} {{- range $ring }} {{- $px := index . 0 }} {{- $py := index . 1 }} {{- $d := add (mul (sub $px $x) (sub $px $x)) (mul (sub $py $y) (sub $py $y)) }} {{- if lt $d $bestDist }} {{- $bestDist = $d }} {{- $bestRing = $ring }} {{- end }} {{- end }} {{- end }} {{- return $bestRing -}}
